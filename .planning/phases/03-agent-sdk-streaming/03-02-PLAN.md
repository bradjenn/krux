**2. Create frontend/src/stores/executionStore.ts:**

Zustand store for execution state. Does NOT use `persist` middleware (execution state is ephemeral -- lost on page reload is fine for Phase 3; session persistence is Phase 4).

```typescript
import { create } from 'zustand'
import type { ExecutionProgress } from '@shared/types'

type ExecutionStatus = 'idle' | 'starting' | 'running' | 'done' | 'error' | 'cancelled'

interface PlanSegment {
  status: 'done' | 'running' | 'pending'
}

interface ExecutionState {
  // Session tracking
  sessionId: string | null
  phaseId: string | null
  phaseName: string | null
  status: ExecutionStatus

  // Progress
  progress: ExecutionProgress | null
  planSegments: PlanSegment[]

  // Error info
  errorMessage: string | null

  // Actions
  startExecution: (sessionId: string, phaseId: string, phaseName: string, planTotal: number) => void
  setRunning: () => void
  updateProgress: (progress: ExecutionProgress) => void
  setComplete: () => void
  setError: (reason: 'cancelled' | 'error', error?: string) => void
  reset: () => void
}

export const useExecutionStore = create<ExecutionState>()((set, get) => ({
  sessionId: null,
  phaseId: null,
  phaseName: null,
  status: 'idle',
  progress: null,
  planSegments: [],
  errorMessage: null,

  startExecution: (sessionId, phaseId, phaseName, planTotal) => set({
    sessionId,
    phaseId,
    phaseName,
    status: 'starting',
    progress: null,
    planSegments: Array.from({ length: planTotal }, () => ({ status: 'pending' as const })),
    errorMessage: null,
  }),

  setRunning: () => set({ status: 'running' }),

  updateProgress: (progress) => {
    const segments = [...get().planSegments]
    // Mark completed plans
    for (let i = 0; i < progress.planCurrent - 1 && i < segments.length; i++) {
      segments[i] = { status: 'done' }
    }
    // Mark current plan as running
    if (progress.planCurrent > 0 && progress.planCurrent <= segments.length) {
      segments[progress.planCurrent - 1] = { status: 'running' }
    }
    set({ progress, planSegments: segments })
  },

  setComplete: () => {
    const segments = get().planSegments.map(() => ({ status: 'done' as const }))
    set({ status: 'done', planSegments: segments })
  },

  setError: (reason, error) => set({
    status: reason === 'cancelled' ? 'cancelled' : 'error',
    errorMessage: error ?? null,
  }),

  reset: () => set({
    sessionId: null,
    phaseId: null,
    phaseName: null,
    status: 'idle',
    progress: null,
    planSegments: [],
    errorMessage: null,
  }),
}))
```

**3. Create frontend/src/hooks/useExecution.ts:**

Hook that provides `startExecution(phaseId, phaseName, planTotal)` and `stopExecution()` functions. These call the backend API.

```typescript
import { useCallback } from 'react'
import { useAppStore } from '@/stores/appStore'
import { useExecutionStore } from '@/stores/executionStore'
import { apiFetch } from '@/lib/api'

export function useExecution() {
  const { activeProjectId } = useAppStore()
  const { sessionId, status } = useExecutionStore()
  const store = useExecutionStore

  const startExecution = useCallback(async (
    phaseId: string,
    phaseName: string,
    planTotal: number,
  ) => {
    if (!activeProjectId) return

    store.getState().startExecution('pending', phaseId, phaseName, planTotal)

    try {
      const result = await apiFetch<{ sessionId: string }>(`/execution/${phaseId}`, {
        method: 'POST',
        body: JSON.stringify({ projectId: activeProjectId }),
      })

      store.getState().startExecution(result.sessionId, phaseId, phaseName, planTotal)
      store.getState().setRunning()
    } catch (err) {
      store.getState().setError('error', err instanceof Error ? err.message : 'Failed to start execution')
    }
  }, [activeProjectId])

  const stopExecution = useCallback(async () => {
    const currentSessionId = store.getState().sessionId
    if (!currentSessionId) return

    try {
      await apiFetch(`/execution/${currentSessionId}`, { method: 'DELETE' })
      // The WS execution:end event will update the store -- no need to update here
    } catch (err) {
      console.error('Failed to stop execution:', err)
    }
  }, [])

  const restartExecution = useCallback(async (
    phaseId: string,
    phaseName: string,
    planTotal: number,
  ) => {
    // Reset store and start fresh (no resume: -- per Pitfall 7 in RESEARCH.md)
    store.getState().reset()
    await startExecution(phaseId, phaseName, planTotal)
  }, [startExecution])

  return {
    startExecution,
    stopExecution,
    restartExecution,
    sessionId,
    status,
    isRunning: status === 'running' || status === 'starting',
  }
}
```

**3. Create frontend/src/components/execution/LogViewer.tsx:**

Scrollable ANSI log viewer with rAF buffering. Per CONTEXT.md decisions:

- "Jump to bottom" button appears when not at bottom
- Distinct terminal feel: darker background, border glow while running

**IMPORTANT SECURITY NOTE about `dangerouslySetInnerHTML`:** The `ansi_up` library HTML-escapes ALL content before processing ANSI codes. Angle brackets become `&lt;` and `&gt;`, quotes are escaped, etc. This is documented behavior -- the output is safe to inject as innerHTML because XSS vectors are neutralized before ANSI conversion. This is NOT raw user input being set as HTML.

```tsx
import { useEffect, useRef, useState, useCallback } from 'react'
import AnsiUp from 'ansi_up'
import { ChevronDown } from 'lucide-react'
import { onExecutionMessage } from '@/hooks/useWebSocket'
import { useExecutionStore } from '@/stores/executionStore'

const ansiUp = new AnsiUp()

export function LogViewer() {
  const [lines, setLines] = useState<string[]>([])
  const bufferRef = useRef<string[]>([])
  const rafRef = useRef<number>(0)
  const containerRef = useRef<HTMLDivElement>(null)
  const bottomRef = useRef<HTMLDivElement>(null)
  const [autoScroll, setAutoScroll] = useState(true)
  const status = useExecutionStore((s) => s.status)

  const isActive = status === 'running' || status === 'starting'

  // Subscribe to execution messages via module-level listener
  useEffect(() => {
    const unsubscribe = onExecutionMessage((text) => {
      // ansi_up.ansi_to_html() HTML-escapes all content before ANSI processing
      // This makes the output XSS-safe (angle brackets, quotes are escaped first)
      const html = ansiUp.ansi_to_html(text)
      bufferRef.current.push(html)
    })
    return unsubscribe
  }, [])

  // rAF drain loop: flush buffer to React state at ~60fps
  useEffect(() => {
    function flush() {
      if (bufferRef.current.length > 0) {
        const batch = bufferRef.current.splice(0)
        setLines((prev) => {
          const updated = [...prev, ...batch]
          // Cap at 10,000 lines (rolling buffer)
          if (updated.length > 10000) {
            return updated.slice(updated.length - 10000)
          }
          return updated
        })
      }
      rafRef.current = requestAnimationFrame(flush)
    }
    rafRef.current = requestAnimationFrame(flush)
    return () => cancelAnimationFrame(rafRef.current)
  }, [])

  // Auto-scroll to bottom when new lines arrive
  useEffect(() => {
    if (autoScroll && bottomRef.current) {
      bottomRef.current.scrollIntoView({ behavior: 'instant' })
    }
  }, [lines, autoScroll])

  // Detect scroll position for auto-scroll toggle
  const handleScroll = useCallback(() => {
    const el = containerRef.current
    if (!el) return
    const atBottom = el.scrollHeight - el.scrollTop - el.clientHeight < 50
    setAutoScroll(atBottom)
  }, [])

  const jumpToBottom = useCallback(() => {
    setAutoScroll(true)
    bottomRef.current?.scrollIntoView({ behavior: 'instant' })
  }, [])

  // Reset lines when a new execution starts
  const sessionId = useExecutionStore((s) => s.sessionId)
  useEffect(() => {
    setLines([])
    bufferRef.current = []
  }, [sessionId])

  return (
    <div className="relative flex-1 overflow-hidden">
      <div
        ref={containerRef}
        className="h-full overflow-y-auto p-4 font-mono text-xs leading-relaxed"
        style={{
          background: '#0a0a0c',
          color: 'var(--color-text2)',
          boxShadow: isActive ? 'inset 0 0 30px rgba(200, 255, 0, 0.03)' : 'none',
          transition: 'box-shadow 0.3s ease',
        }}
        onScroll={handleScroll}
      >
        {lines.length === 0 && status === 'starting' && (
          <div className="flex items-center gap-2" style={{ color: 'var(--color-text3)' }}>
            <div className="w-2 h-2 rounded-full animate-pulse" style={{ background: 'var(--color-accent)' }} />
            Starting agent... this may take up to 15 seconds on first run.
          </div>
        )}
        {lines.map((htmlLine, i) => (
          // SAFE: ansi_up HTML-escapes all content before ANSI code processing.
          // Angle brackets, quotes, and other HTML-special chars are escaped first.
          // See: https://github.com/drudru/ansi_up#security
          <div
            key={i}
            className="whitespace-pre-wrap break-all"
            dangerouslySetInnerHTML={{ __html: htmlLine }}
          />
        ))}
        <div ref={bottomRef} />
      </div>

      {/* Jump to bottom button */}
      {!autoScroll && lines.length > 0 && (
        <button
          onClick={jumpToBottom}
          className="absolute bottom-4 right-4 flex items-center gap-1 text-xs px-3 py-1.5 rounded-full font-medium shadow-lg transition-opacity duration-200"
          style={{
            background: 'var(--color-bg3)',
            color: 'var(--color-text2)',
```
