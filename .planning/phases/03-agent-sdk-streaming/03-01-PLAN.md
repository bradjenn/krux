**2. Create backend/src/services/agentRunner.ts:**

This is the core service that wraps the SDK `query()` function. Key design:
- Fire-and-forget: `startAgentExecution()` returns void, runs async in background
- AbortController created before query, stored in SessionRegistry
- Session ID captured from `system:init` message for warm resume
- All SDK messages broadcast as WS events
- Progress parsing extracts plan/task counts from assistant message text

```typescript
import { query } from '@anthropic-ai/claude-agent-sdk'
import { broadcast } from '../routes/ws.js'
import { SessionRegistry } from './sessionRegistry.js'

export function startAgentExecution(
  sessionId: string,
  projectPath: string,
  phasePrompt: string,
  resumeSessionId?: string,
): void {
  const ac = new AbortController()
  SessionRegistry.setAbortController(sessionId, ac)
  SessionRegistry.update(sessionId, { status: 'running' })

  // Broadcast started event
  const session = SessionRegistry.get(sessionId)
  broadcast({
    type: 'execution:started',
    sessionId,
    phaseId: session?.phaseId ?? '',
  })

  // Fire-and-forget async execution
  void (async () => {
    try {
      const q = query({
        prompt: phasePrompt,
        options: {
          cwd: projectPath,
          abortController: ac,
          permissionMode: 'bypassPermissions',
          allowedTools: ['Read', 'Write', 'Edit', 'Bash', 'Glob', 'Grep'],
          ...(resumeSessionId ? { resume: resumeSessionId } : {}),
        },
      })

      for await (const message of q) {
        // Capture SDK session ID from init message for warm resume
        if (message.type === 'system' && message.subtype === 'init') {
          SessionRegistry.update(sessionId, { sdkSessionId: message.session_id })
        }

        // Extract text content from assistant messages for streaming
        if (message.type === 'assistant') {
          const textContent = message.message.content
            .filter((b: { type: string }) => b.type === 'text')
            .map((b: { type: string; text?: string }) => b.text ?? '')
            .join('')

          if (textContent) {
            broadcast({ type: 'execution:message', sessionId, text: textContent })

            // Parse progress from text content
            const progress = parseProgress(textContent)
            if (progress) {
              broadcast({ type: 'execution:progress', sessionId, progress })
            }
          }
        }

        // Also stream result messages
        if (message.type === 'result') {
          if (message.subtype === 'error' && message.error) {
            broadcast({
              type: 'execution:message',
              sessionId,
              text: `Error: ${message.error}`,
            })
          }
        }
      }

      SessionRegistry.update(sessionId, { status: 'done' })
      broadcast({ type: 'execution:complete', sessionId })
    } catch (err) {
      const reason = ac.signal.aborted ? 'cancelled' : 'error'
      const errorMsg = err instanceof Error ? err.message : String(err)
      SessionRegistry.update(sessionId, {
        status: reason === 'cancelled' ? 'cancelled' : 'error',
        error: errorMsg,
      })
      broadcast({ type: 'execution:end', sessionId, reason, error: errorMsg })
    }
  })()
}

// Parse plan/task progress from GSD agent output text.
// Looks for patterns like "Plan 2/4: Agent Service" and "Task 3/8: Create session"
// Note: exact format may vary -- this is a best-effort regex that should be validated
// against real GSD agent output and adjusted if needed.
function parseProgress(text: string): import('../../shared/types/index.js').ExecutionProgress | null {
  const planMatch = text.match(/Plan\s+(\d+)\s*\/\s*(\d+)\s*:\s*(.+?)(?:\s*[•|—-]|$)/m)
  const taskMatch = text.match(/Task\s+(\d+)\s*\/\s*(\d+)\s*:\s*(.+?)(?:\n|$)/m)
  if (!planMatch && !taskMatch) return null
  return {
    planCurrent: planMatch ? parseInt(planMatch[1], 10) : 0,
    planTotal:   planMatch ? parseInt(planMatch[2], 10) : 0,
    planName:    planMatch ? planMatch[3].trim() : '',
    taskCurrent: taskMatch ? parseInt(taskMatch[1], 10) : 0,
    taskTotal:   taskMatch ? parseInt(taskMatch[2], 10) : 0,
    taskName:    taskMatch ? taskMatch[3].trim() : '',
  }
}
```

**3. Create backend/src/routes/execution.ts:**

Two endpoints:
- `POST /:phaseId` -- starts execution, returns session ID immediately
- `DELETE /:sessionId` -- stops a running execution via AbortController

Follow the existing Hono route pattern from sessions.ts:

```typescript
import { Hono } from 'hono'
import { SessionRegistry } from '../services/sessionRegistry.js'
import { ProjectStore } from '../services/projectStore.js'
import { startAgentExecution } from '../services/agentRunner.js'
import type { ApiError } from '../types/index.js'

export const executionRoute = new Hono()

// POST /:phaseId — start phase execution
executionRoute.post('/:phaseId', async (c) => {
  let body: { projectId?: string; resumeSessionId?: string } = {}
  try {
    body = await c.req.json<{ projectId?: string; resumeSessionId?: string }>()
  } catch {
    return c.json<ApiError>(
      { error: { code: 'BAD_REQUEST', message: 'Request body must be valid JSON' } },
      400,
    )
  }

  const phaseId = c.req.param('phaseId')
  const { projectId, resumeSessionId } = body

  if (!projectId) {
    return c.json<ApiError>(
      { error: { code: 'BAD_REQUEST', message: 'projectId is required' } },
      400,
    )
  }

  const project = ProjectStore.get(projectId)
  if (!project) {
    return c.json<ApiError>(
      { error: { code: 'NOT_FOUND', message: `Project '${projectId}' not found` } },
      404,
    )
  }

  // Check for already-running execution on this project
  const existingSessions = SessionRegistry.listByProject(projectId)
  const runningSession = existingSessions.find(s => s.status === 'running')
  if (runningSession) {
    return c.json<ApiError>(
      { error: { code: 'CONFLICT', message: 'An execution is already running for this project' } },
      409,
    )
  }

  // Create session and start execution
  const session = SessionRegistry.create(projectId, phaseId)

  // Build the phase execution prompt
  // This tells Claude to execute the GSD phase plans
  const phasePrompt = `Execute GSD phase ${phaseId} for the project at ${project.path}. Read the phase plans in .planning/phases/ and execute them in order. Follow the GSD execution workflow.`

  startAgentExecution(session.id, project.path, phasePrompt, resumeSessionId)

  return c.json({ sessionId: session.id, phaseId, status: 'running' }, 201)
})

// DELETE /:sessionId — stop a running execution
executionRoute.delete('/:sessionId', (c) => {
  const sessionId = c.req.param('sessionId')
  const session = SessionRegistry.get(sessionId)

  if (!session) {
    return c.json<ApiError>(
      { error: { code: 'NOT_FOUND', message: `Session '${sessionId}' not found` } },
      404,
    )
  }

  if (session.status !== 'running') {
    return c.json<ApiError>(
      { error: { code: 'BAD_REQUEST', message: 'Session is not running' } },
      400,
    )
  }

  const ac = SessionRegistry.getAbortController(sessionId)
  if (ac) {
    ac.abort()
  }

  return new Response(null, { status: 204 })
})

// GET /:sessionId — get session status (useful for polling fallback)
executionRoute.get('/:sessionId', (c) => {
  const sessionId = c.req.param('sessionId')
  const session = SessionRegistry.get(sessionId)
  if (!session) {
    return c.json<ApiError>(
      { error: { code: 'NOT_FOUND', message: `Session '${sessionId}' not found` } },
      404,
    )
  }
  return c.json(session)
})
```
