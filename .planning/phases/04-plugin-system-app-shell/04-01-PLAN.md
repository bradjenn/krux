    ```typescript
    import type { PluginDefinition, ViewDefinition } from './types'

    export function buildViewRegistry(plugins: PluginDefinition[]): Map<string, ViewDefinition> {
      const map = new Map<string, ViewDefinition>()
      for (const plugin of plugins) {
        for (const view of plugin.views) {
          map.set(view.id, view)
        }
      }
      return map
    }

    export function getDefaultViewId(plugins: PluginDefinition[]): string | null {
      for (const plugin of plugins) {
        const defaultView = plugin.views.find((v) => v.isDefault)
        if (defaultView) return defaultView.id
      }
      return plugins[0]?.views[0]?.id ?? null
    }

    export function getActivePlugin(plugins: PluginDefinition[], viewId: string | null): PluginDefinition | null {
      if (!viewId) return null
      return plugins.find((p) => p.views.some((v) => v.id === viewId)) ?? null
    }
    ```

    ```typescript
    export type AppEvents = {
      'project:switched': { projectId: string | null }
      'file:changed': { projectId: string; path: string }
      'execution:started': { sessionId: string; phaseId: string }
    }

    export const appEvents = mitt<AppEvents>()

    export function onAppEvent<K extends keyof AppEvents>(
      event: K,
      handler: (payload: AppEvents[K]) => void
    ): () => void {
      appEvents.on(event, handler)
      return () => appEvents.off(event, handler)
    }
    ```

      Define two handler functions outside the hook:
      ```typescript
      function handleExecutionEvent(event: ExecutionWsEvent): void {
        const store = useExecutionStore.getState()
        switch (event.type) {
          case 'execution:started':
            store.setRunning()
            appEvents.emit('execution:started', { sessionId: event.sessionId, phaseId: event.phaseId })
            break
          case 'execution:message':
            for (const listener of executionMessageListeners) {
              listener(event.text)
            }
            break
          case 'execution:progress':
            store.updateProgress(event.progress)
            break
          case 'execution:complete':
            store.setComplete()
            toast.success('Phase execution complete')
            break
          case 'execution:end':
            store.setError(event.reason, event.error)
            if (event.reason === 'error') {
              toast.error('Phase execution failed', { description: event.error })
            } else {
              toast('Execution stopped')
            }
            break
        }
      }

      function handleFileEvent(event: FileWsEvent, queryClient: ReturnType<typeof useQueryClient>): void {
        const { projectId, path, type } = event
        queryClient.invalidateQueries({ queryKey: ['filetree', projectId], refetchType: 'active' })
        queryClient.invalidateQueries({ queryKey: ['overview', projectId], refetchType: 'active' })
        appEvents.emit('file:changed', { projectId, path })

        if (type === 'file:changed') {
          queryClient.invalidateQueries({ queryKey: ['doc', projectId, path], refetchType: 'active' })
          recentChanges.set(path, Date.now())
        }
        if (type === 'file:created') {
          const filename = path.split('/').pop() ?? path
          toast('New file', { description: filename, duration: 4000 })
          recentChanges.set(path, Date.now())
        }
        if (type === 'file:deleted') {
          queryClient.invalidateQueries({ queryKey: ['doc', projectId, path], refetchType: 'active' })
        }
      }
      ```

      Then in `ws.onmessage`:
      ```typescript
      ws.onmessage = (evt) => {
        if (unmounted.current) return
        try {
          const event: WsEvent = JSON.parse(evt.data as string)
          const namespace = event.type.split(':')[0]
          if (namespace === 'execution') {
            handleExecutionEvent(event as ExecutionWsEvent)
          } else if (namespace === 'file') {
            handleFileEvent(event as FileWsEvent, queryClient)
          }
        } catch (e) {
          console.error('Failed to parse WS message:', e)
        }
      }
      ```
