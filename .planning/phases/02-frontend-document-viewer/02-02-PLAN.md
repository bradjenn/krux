    **Step 1: Add GET /:id/docs directory listing to the backend**

    In `backend/src/routes/docs.ts`, add a NEW route handler for `GET /:id/docs` (no wildcard -- this returns the directory tree as JSON). This route MUST be registered BEFORE the existing `GET /:id/docs/*` wildcard route so Hono matches it first.

    The endpoint:
    1. Validates the project exists via ProjectStore.get(id)
    2. Reads the `.planning/` directory recursively using `readdirSync` with `{ withFileTypes: true }`
    3. Returns a nested JSON tree with the `TreeNode` shape:
       ```typescript
       interface TreeNode {
         name: string
         path: string         // relative to .planning/
         type: 'file' | 'dir'
         children?: TreeNode[]
         status?: 'complete' | 'in-progress' | undefined
       }
       ```
    4. Sorting rules:
       - Top-level: non-phase files first (ROADMAP.md, STATE.md, PROJECT.md, REQUIREMENTS.md, etc.), then directories (phases/, codebase/, etc.)
       - Within each directory level: directories first, then files, alphabetically
    5. Status detection for PLAN files:
       - If a `-PLAN.md` file has a corresponding `-SUMMARY.md` file in the same directory -> `status: 'complete'`
       - If a `-PLAN.md` file exists without a corresponding `-SUMMARY.md` -> `status: 'in-progress'`
       - All other files -> `status: undefined`

    Implementation helper function:
    ```typescript
    function buildTree(dir: string, rootDir: string): TreeNode[] {
      const entries = readdirSync(dir, { withFileTypes: true })
      const siblings = entries.map(e => e.name) // for status detection

      const nodes: TreeNode[] = []
      for (const entry of entries) {
        const relPath = join(dir, entry.name).slice(rootDir.length + 1) // +1 for trailing /
        if (entry.isDirectory()) {
          nodes.push({
            name: entry.name,
            path: relPath,
            type: 'dir',
            children: buildTree(join(dir, entry.name), rootDir),
          })
        } else {
          const status = detectPlanStatus(entry.name, siblings)
          nodes.push({ name: entry.name, path: relPath, type: 'file', status })
        }
      }
      // Sort: dirs first, then files, alphabetically within each group
      return nodes.sort((a, b) => {
        if (a.type !== b.type) return a.type === 'dir' ? -1 : 1
        return a.name.localeCompare(b.name)
      })
    }

    function detectPlanStatus(filename: string, siblings: string[]): 'complete' | 'in-progress' | undefined {
      const planMatch = filename.match(/^(.+-\d+)-PLAN\.md$/)
      if (!planMatch) return undefined
      const summaryName = `${planMatch[1]}-SUMMARY.md`
      return siblings.includes(summaryName) ? 'complete' : 'in-progress'
    }
    ```

    **Step 2: Create the file tree TanStack Query hook**

    `frontend/src/hooks/useFileTree.ts`:
    ```typescript
    import { useQuery } from '@tanstack/react-query'
    import { apiFetch } from '@/lib/api'

    export interface TreeNode {
      name: string
      path: string
      type: 'file' | 'dir'
      children?: TreeNode[]
      status?: 'complete' | 'in-progress'
    }

    export function useFileTree(projectId: string | null) {
      return useQuery<TreeNode[]>({
        queryKey: ['filetree', projectId],
        queryFn: () => apiFetch<TreeNode[]>(`/projects/${projectId}/docs`),
        enabled: !!projectId,
        staleTime: 30_000,
      })
    }
    ```

    export function useDocument(projectId: string | null, filePath: string | null) {

    ```tsx
    import ReactMarkdown from 'react-markdown'
    import remarkGfm from 'remark-gfm'
    import rehypeHighlight from 'rehype-highlight'
    import 'highlight.js/styles/atom-one-dark.css'

    const markdownComponents = {
      // Headings: all JetBrains Mono (inherited), accent for h1
      h1: ({ children }) => <h1 className="text-2xl font-bold text-[var(--color-accent)] mb-4 mt-6 first:mt-0">{children}</h1>,
      h2: ({ children }) => <h2 className="text-xl font-semibold text-[var(--color-text)] mb-3 mt-5">{children}</h2>,
      h3: ({ children }) => <h3 className="text-lg font-semibold text-[var(--color-text)] mb-2 mt-4">{children}</h3>,
      h4: ({ children }) => <h4 className="text-base font-semibold text-[var(--color-text2)] mb-2 mt-3">{children}</h4>,

      // Paragraphs
      p: ({ children }) => <p className="text-sm text-[var(--color-text)] mb-3 leading-relaxed">{children}</p>,

      // Lists
      ul: ({ children }) => <ul className="list-disc list-inside text-sm text-[var(--color-text)] mb-3 space-y-1">{children}</ul>,
      ol: ({ children }) => <ol className="list-decimal list-inside text-sm text-[var(--color-text)] mb-3 space-y-1">{children}</ol>,
      li: ({ children }) => <li className="text-sm">{children}</li>,

      // Tables (GFM)
      table: ({ children }) => <div className="overflow-x-auto mb-4"><table className="border-collapse w-full text-sm">{children}</table></div>,
      thead: ({ children }) => <thead className="bg-[var(--color-bg3)]">{children}</thead>,
      th: ({ children }) => <th className="border border-[var(--color-border2)] px-3 py-1.5 text-left font-semibold text-[var(--color-text2)]">{children}</th>,
      td: ({ children }) => <td className="border border-[var(--color-border)] px-3 py-1.5 text-[var(--color-text3)]">{children}</td>,

      // Task list checkboxes (remark-gfm)
      input: ({ checked, ...props }) => (
        <input type="checkbox" checked={checked} readOnly className="mr-2 accent-[var(--color-accent)]" {...props} />
      ),

      // Code blocks (rehype-highlight handles syntax, but we style the container)
      pre: ({ children }) => <pre className="bg-[var(--color-bg)] border border-[var(--color-border)] rounded-[10px] p-4 mb-4 overflow-x-auto text-sm">{children}</pre>,
      code: ({ className, children, ...props }) => {
        // Inline code vs block code
        const isBlock = className?.startsWith('hljs') || className?.startsWith('language-')
        if (isBlock) return <code className={className} {...props}>{children}</code>
        return <code className="bg-[var(--color-bg3)] px-1.5 py-0.5 rounded text-[var(--color-accent)] text-sm" {...props}>{children}</code>
      },

      // Blockquotes
      blockquote: ({ children }) => <blockquote className="border-l-2 border-[var(--color-accent)] pl-4 my-3 text-[var(--color-text2)]">{children}</blockquote>,

      // Horizontal rule
      hr: () => <hr className="border-[var(--color-border)] my-6" />,

      // Links
      a: ({ href, children }) => <a href={href} className="text-[var(--color-accent)] underline hover:text-[var(--color-accent2)]" target="_blank" rel="noopener noreferrer">{children}</a>,

      // Strong / Em
      strong: ({ children }) => <strong className="font-bold text-[var(--color-text)]">{children}</strong>,
      em: ({ children }) => <em className="italic text-[var(--color-text2)]">{children}</em>,
    }

    export function DocViewer({ content }: { content: string }) {
      return (
        <div className="max-w-4xl">
          <ReactMarkdown
            remarkPlugins={[remarkGfm]}
            rehypePlugins={[rehypeHighlight]}
            components={markdownComponents}
          >
            {content}
          </ReactMarkdown>
        </div>
      )
    }
    ```

    **IMPORTANT:** Do NOT add `rehype-raw`. react-markdown renders safely by default. Adding rehype-raw enables raw HTML passthrough and introduces risk.

    **Step 4: Create the Breadcrumb component**

    `frontend/src/components/viewer/Breadcrumb.tsx`:
    - Shows path context for the currently viewed file
    - Format: `phases > 01-backend-foundation > 01-01-PLAN.md`
    - Split the file path by `/` and render each segment as a clickable breadcrumb
    - Clicking a directory segment navigates to that directory (sets selectedFile to the directory path)
    - Last segment (current file) is highlighted with accent color, non-clickable
    - Uses Lucide ChevronRight as separator
    - Style: text-xs, text3 for non-active segments, accent for active segment

    ```tsx
    export function Breadcrumb({ filePath }: { filePath: string }) {
      const setSelectedFile = useAppStore((s) => s.setSelectedFile)
      const segments = filePath.split('/')

      return (
        <nav className="flex items-center gap-1 text-xs mb-4 pb-3 border-b border-[var(--color-border)]">
          {segments.map((segment, i) => {
            const isLast = i === segments.length - 1
            const path = segments.slice(0, i + 1).join('/')
            return (
              <Fragment key={path}>
                {i > 0 && <ChevronRight size={12} className="text-[var(--color-text4)]" />}
                {isLast ? (
                  <span className="text-[var(--color-accent)] font-semibold">{segment}</span>
                ) : (
                  <button onClick={() => setSelectedFile(path)} className="text-[var(--color-text3)] hover:text-[var(--color-text)]">{segment}</button>
                )}
              </Fragment>
            )
          })}
        </nav>
      )
    }
    ```

    export function DocumentPage() {
      const activeProjectId = useAppStore((s) => s.activeProjectId)
      const selectedFile = useAppStore((s) => s.selectedFile)
      const { data: tree, isLoading: treeLoading } = useFileTree(activeProjectId)
      const { data: content, isLoading: docLoading } = useDocument(
        activeProjectId,
        selectedFile?.endsWith('/') ? null : selectedFile  // don't fetch directories
      )

      return (
        <div className="flex h-full">
          {/* File tree panel */}
          <div className="w-[280px] flex-shrink-0 border-r border-[var(--color-border)] bg-[var(--color-bg2)]">
            <ScrollArea className="h-full">
              {treeLoading ? <TreeSkeleton /> : tree && <FileTree nodes={tree} />}
            </ScrollArea>
          </div>
          {/* Document content panel */}
          <div className="flex-1 overflow-auto p-6">
            {selectedFile && !selectedFile.endsWith('/') ? (
              <>
                <Breadcrumb filePath={selectedFile} />
                {docLoading ? <DocSkeleton /> : content ? <DocViewer content={content} /> : <FileNotFound />}
              </>
            ) : (
              <div className="flex items-center justify-center h-full text-[var(--color-text3)]">
                <div className="text-center">
                  <FileText size={48} className="mx-auto mb-3 opacity-50" />
                  <p className="text-sm">Select a file from the tree</p>
                </div>
              </div>
            )}
          </div>
        </div>
      )
    }
