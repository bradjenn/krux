    **Step 2: Create the watcher service**

    `backend/src/lib/watcher.ts`:

    ```typescript
    import chokidar, { type FSWatcher } from 'chokidar'
    import { join } from 'node:path'
    import { ProjectStore } from '../services/projectStore.js'

    export type WsEvent = {
      type: 'file:changed' | 'file:created' | 'file:deleted'
      projectId: string
      path: string  // relative to .planning/
    }

    let watcher: FSWatcher | null = null

    export function createWatcher(broadcast: (event: WsEvent) => void): FSWatcher | null {
      const projects = ProjectStore.list()
      const paths = projects.map((p) => join(p.path, '.planning'))

      if (paths.length === 0) {
        // No projects to watch -- return null, will be created when projects are added
        return null
      }

      watcher = chokidar.watch(paths, {
        ignoreInitial: true,
        persistent: true,
        awaitWriteFinish: { stabilityThreshold: 100, pollInterval: 50 },
      })

      watcher.on('change', (filePath) => {
        const event = resolveEvent('file:changed', filePath, projects)
        if (event) broadcast(event)
      })
      watcher.on('add', (filePath) => {
        const event = resolveEvent('file:created', filePath, projects)
        if (event) broadcast(event)
      })
      watcher.on('unlink', (filePath) => {
        const event = resolveEvent('file:deleted', filePath, projects)
        if (event) broadcast(event)
      })

      return watcher
    }

    // Add a new project path to the existing watcher, or create a new watcher if none exists
    export function addWatchPath(projectPath: string, broadcast: (event: WsEvent) => void): void {
      const planningPath = join(projectPath, '.planning')
      if (watcher) {
        watcher.add(planningPath)
      } else {
        // First project -- create the watcher
        createWatcher(broadcast)
      }
    }

    // Remove a project path from the watcher
    export function removeWatchPath(projectPath: string): void {
      if (!watcher) return
      const planningPath = join(projectPath, '.planning')
      watcher.unwatch(planningPath)
    }

    // Graceful shutdown
    export function closeWatcher(): Promise<void> {
      if (!watcher) return Promise.resolve()
      return watcher.close()
    }

    function resolveEvent(
      type: WsEvent['type'],
      filePath: string,
      projects: Array<{ id: string; path: string }>
    ): WsEvent | null {
      // Refresh project list to catch newly added projects
      const currentProjects = projects.length > 0 ? projects : ProjectStore.list()
      for (const p of currentProjects) {
        const planningDir = join(p.path, '.planning')
        // Normalize: ensure we check with trailing separator
        if (filePath.startsWith(planningDir + '/') || filePath.startsWith(planningDir + '\\')) {
          return { type, projectId: p.id, path: filePath.slice(planningDir.length + 1) }
        }
      }
      return null
    }
    ```

    **Step 3: Create the WebSocket route**

    `backend/src/routes/ws.ts`:

    This module exports a function that takes the Hono app and returns the `injectWebSocket` function (to be called after `serve()`).

    ```typescript
    import { createNodeWebSocket } from '@hono/node-ws'
    import type { Hono } from 'hono'
    import type { WsEvent } from '../lib/watcher.js'

    // Set of connected clients for broadcasting
    const clients = new Set<import('@hono/node-ws').WSContext>()

    export function setupWebSocket(app: Hono) {
      const { injectWebSocket, upgradeWebSocket } = createNodeWebSocket({ app })

      // Register WS route BEFORE any CORS middleware would affect it
      // The /ws path is outside /api/* scope, so CORS doesn't apply
      app.get('/ws', upgradeWebSocket(() => ({
        onOpen(_evt, ws) {
          clients.add(ws)
          console.log(`WebSocket client connected (${clients.size} total)`)
        },
        onClose(_evt, ws) {
          clients.delete(ws)
          console.log(`WebSocket client disconnected (${clients.size} remaining)`)
        },
        onError(evt, ws) {
          console.error('WebSocket error:', evt)
          clients.delete(ws)
        },
      })))

      return { injectWebSocket }
    }

    // Broadcast a WS event to all connected clients
    export function broadcast(event: WsEvent): void {
      const msg = JSON.stringify(event)
      for (const ws of clients) {
        try {
          ws.send(msg)
        } catch {
          // Client disconnected, will be cleaned up on next onClose
          clients.delete(ws)
        }
      }
    }
    ```

    **Step 1: Create the WebSocket client hook**

    `frontend/src/hooks/useWebSocket.ts`:

    Hand-roll a simple WebSocket hook (do NOT use react-use-websocket library). Includes:
    - Auto-connect on mount
    - Exponential backoff reconnection: 1s -> 2s -> 4s -> 8s cap
    - Parse incoming WS messages as WsEvent
    - Invalidate TanStack Query cache based on event type
    - Track connection status for "Disconnected" banner
    - Emit custom events for UI effects (toast, highlight)

    ```typescript
    import { useEffect, useRef, useCallback, useState } from 'react'
    import { useQueryClient } from '@tanstack/react-query'
    import { toast } from 'sonner'

    type WsEvent = {
      type: 'file:changed' | 'file:created' | 'file:deleted'
      projectId: string
      path: string
    }

    type ConnectionStatus = 'connecting' | 'connected' | 'disconnected'

    // Track recently changed files for highlight glow effect
    // Map from file path -> timestamp, cleared after animation duration
    const recentChanges = new Map<string, number>()

    export function useRecentChange(filePath: string): boolean {
      // Hook that returns true if this file was recently changed
      const [changed, setChanged] = useState(false)
      useEffect(() => {
        const ts = recentChanges.get(filePath)
        if (ts && Date.now() - ts < 2000) {
          setChanged(true)
          const timer = setTimeout(() => {
            setChanged(false)
            recentChanges.delete(filePath)
          }, 2000)
          return () => clearTimeout(timer)
        }
      }, [filePath])
      return changed
    }

    export function useWebSocket() {
      const queryClient = useQueryClient()
      const wsRef = useRef<WebSocket | null>(null)
      const reconnectAttempt = useRef(0)
      const [status, setStatus] = useState<ConnectionStatus>('connecting')

      const connect = useCallback(() => {
        const ws = new WebSocket('ws://localhost:3001/ws')
        wsRef.current = ws

        ws.onopen = () => {
          setStatus('connected')
          reconnectAttempt.current = 0
          console.log('WebSocket connected')
        }

        ws.onmessage = (evt) => {
          try {
            const event: WsEvent = JSON.parse(evt.data)
            const { projectId, path, type } = event

            // Always invalidate file tree and overview
            queryClient.invalidateQueries({ queryKey: ['filetree', projectId], refetchType: 'active' })
            queryClient.invalidateQueries({ queryKey: ['overview', projectId], refetchType: 'active' })

            if (type === 'file:changed') {
              // Invalidate the specific document content
              queryClient.invalidateQueries({ queryKey: ['doc', projectId, path], refetchType: 'active' })
              // Track for highlight glow
              recentChanges.set(path, Date.now())
            }

            if (type === 'file:created') {
              // Toast notification for new files
              const filename = path.split('/').pop() ?? path
              toast('New file', {
                description: filename,
                duration: 4000,
              })
              // Also track for highlight
              recentChanges.set(path, Date.now())
            }

            if (type === 'file:deleted') {
              // Invalidate doc if it was the viewed file
              queryClient.invalidateQueries({ queryKey: ['doc', projectId, path], refetchType: 'active' })
            }
          } catch (e) {
            console.error('Failed to parse WS message:', e)
          }
        }

        ws.onclose = () => {
          setStatus('disconnected')
          wsRef.current = null
          // Exponential backoff: 1s, 2s, 4s, 8s cap
          const delay = Math.min(1000 * Math.pow(2, reconnectAttempt.current), 8000)
          reconnectAttempt.current++
          console.log(`WebSocket disconnected, reconnecting in ${delay}ms`)
          setTimeout(connect, delay)
        }

        ws.onerror = () => {
          ws.close()
        }
      }, [queryClient])

      useEffect(() => {
        connect()
        return () => {
          wsRef.current?.close()
        }
      }, [connect])

      return { status }
    }
    ```

    ```tsx
    export function DocumentPage() {
      const scrollRef = useRef<HTMLDivElement>(null)
      const prevScrollTop = useRef(0)
      const prevContent = useRef<string | undefined>()

      // Save scroll position before content changes
      useEffect(() => {
        if (content && content !== prevContent.current) {
          // Content changed -- was it from a WS update?
          if (prevContent.current !== undefined && scrollRef.current) {
            // Restore scroll position after React re-renders
            const savedTop = prevScrollTop.current
            requestAnimationFrame(() => {
              if (scrollRef.current) {
                scrollRef.current.scrollTop = savedTop
              }
            })
          }
          prevContent.current = content
        }
      }, [content])

      // Track scroll position continuously
      const handleScroll = useCallback(() => {
        if (scrollRef.current) {
          prevScrollTop.current = scrollRef.current.scrollTop
        }
      }, [])

      return (
        <div className="flex h-full">
          {/* ... file tree panel ... */}
          <div
            ref={scrollRef}
            className="flex-1 overflow-auto p-6"
            onScroll={handleScroll}
          >
            {/* ... doc viewer ... */}
          </div>
        </div>
      )
    }
    ```
